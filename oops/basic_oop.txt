__dict__
Ye sabse kaam ki cheez hai! Ye object ki Personal Diary hai. Ye batata hai ki "Abhi is waqt"
is object ne apne andar kya values store ki huyi hain.
class Student:
    def __init__(self, name, age):
        self.name = name
        self.age = age

s1 = Student("Anjali", 20)
print(s1.__dict__) 

# Output: {'name': 'Anjali', 'age': 20}
# Dekho, isne bata diya ki s1 ke andar kya data hai!
